CREATE DATABASE imobiliaria;
USE imobiliaria;

CREATE TABLE usuario(
	id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
	usuario varchar(50),
	senha varchar(200)
);

CREATE TABLE locatario(
	id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
	nome varchar(100),
	cpf varchar(14),
	rg varchar(20),
	profissao varchar(50),
	estado_civil varchar(20)
);

CREATE TABLE locador(
	id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
   	nome varchar(100),
	cpf varchar(14),
	rg varchar(20),
	profissao varchar(50),
	estado_civil varchar(20),
	endereco varchar(200)
);

CREATE TABLE imovel(
	id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
	id_locador int, FOREIGN KEY (id_locador) REFERENCES locador(id),
	rua varchar(100),
	numero int,
	bairro varchar(20),
	cidade varchar(50),
	n_agua varchar(20),
	n_luz varchar(20),
	valor float
);

CREATE TABLE contrato(
	id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
	id_locador int, FOREIGN KEY (id_locador) REFERENCES locador(id),
	id_locatario int, FOREIGN KEY (id_locatario) REFERENCES locatario(id),
	id_imovel int, FOREIGN KEY (id_imovel) REFERENCES imovel(id),
	valor_imovel float, FOREIGN KEY (valor_imovel) REFERENCES imovel(valor),
	n_contrato int,
	data_inicio date,
	data_fim date   
);
